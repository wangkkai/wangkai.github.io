>输入一个链表，反转链表后，输出链表的所有元素。

思路：用一个节点pre表示当前节点的前一个节点。用一个节点next表示当前节点的下一个节点。然后将本来指向后一节点的指针指向前一个节点。
```
public class Solution {
    public ListNode ReverseList(ListNode head) {
        if(head==null){
            return null;
        }
        ListNode next=null;
        ListNode pre=null;
        while(head!=null){
            //用next保存原head节点的下一个节点的信息  防止head指向pre后链断裂
            next=head.next;
            //head指向前一个节点
            head.next=pre;
            //pre head一次后移一个节点
            pre=head;
            head=next;
        }
        //head为null 说明pre是最后一个节点 返回pre
        return pre;
    }
}

 class ListNode {
    int val;
    ListNode next = null;

    ListNode(int val) {
        this.val = val;
    }
}
```
