>操作给定的二叉树，将其变换为源二叉树的镜像。

![](http://oc6shen8h.bkt.clouddn.com/%E9%95%9C%E5%83%8F.jpg)

```
public class Solution {
    public void Mirror(TreeNode root) {
        if(root==null){
            return ;
        }
        if(root!=null){
            //交换左右结点
            TreeNode node=null;
            node =root.left;
            root.left=root.right;
            root.right=node;
        }
        //递归交换
        Mirror(root.left);
        Mirror(root.right);
    }
}
public class TreeNode {
    int val = 0;
    TreeNode left = null;
    TreeNode right = null;

    public TreeNode(int val) {
        this.val = val;

    }

}
```
